<script type="text/javascript">
     'use strict';
    var keys=document.querySelectorAll(".key");
    window.onkeydown=function(ev){
         var keyAudio=document.querySelectorAll("audio");
         for(var i=0;i<keyAudio.length;i++){
            if(ev.keyCode==keyAudio[i].attributes["data-key"].nodeValue){

                keyAudio[i].currentTime = 0;
                keyAudio[i].play();

                keys[i].classList.add("playing");

                break;
            }
        }    
    }
    keys.forEach(key => key.addEventListener("transitionend",function(e){
        if(e.propertyName !=='border-left-color') return;
        e.target.classList.remove("playing");
    }))
</script>
